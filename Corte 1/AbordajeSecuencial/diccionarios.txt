### Dict Comprehensions ###
# Dos listas de nombres y alturas
names = ['Jenny', 'Alexus', 'Sam', 'Grace']
heights = [61, 70, 67, 64]

# zip() combina ambas listas en pares (tuplas)
zipStudents = zip(names, heights)
print("zipStudents: ", zipStudents)  # es un objeto iterable

# Crear un diccionario con comprensión: {clave: valor for ...}
students = {key:value for key, value in zip(names, heights)}
# Resultado: {'Jenny': 61, 'Alexus': 70, 'Sam': 67, 'Grace': 64}
print(students)

# Otro ejemplo con bebidas y cafeína
drinks = ["espresso", "chai", "decaf", "drip"]
caffeine = [64, 40, 0, 120]

# Combinar listas en pares
zipped_drinks = zip(drinks, caffeine)
print(zipped_drinks)  # objeto zip

# Crear diccionario {bebida: cafeína}
drinks_to_caffeine = {key:value for key, value in zipped_drinks}
print(drinks_to_caffeine)


# Diccionario de canciones con sus reproducciones
songs = ["Like a Rolling Stone", "Satisfaction", "Imagine", "What's Going On", "Respect", "Good Vibrations"]
playcounts = [78, 29, 44, 21, 89, 5]

# Crear diccionario con zip()
plays = {key:value for key, value in zip(songs, playcounts)}
print(plays)

# Agregar nueva canción
plays.update({"Purple Haze": 1})

# Sobrescribir el valor de "Respect" (antes 89, ahora 94)
plays.update({"Respect": 94})
print("After: ", plays)

# Diccionario que contiene otros diccionarios (anidado)
library = {"The Best Songs": plays, "Sunday Feelings": {}}
print(library)
